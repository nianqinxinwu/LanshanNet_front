### 步骤3: 配置Nginx

#### 3.1 配置文件位置

```bash
/etc/nginx/conf.d/andone.zzlanshan.site.conf
```

#### 3.2 完整Nginx配置

```nginx
# HTTP重定向到HTTPS
server {
    listen 80;
    server_name andone.zzlanshan.site;
    return 301 https://$server_name$request_uri;
}

# HTTPS主配置
server {
    listen 443 ssl http2;
    server_name andone.zzlanshan.site;

    # SSL证书配置
    ssl_certificate /path/to/ssl/andone.zzlanshan.site.crt;
    ssl_certificate_key /path/to/ssl/andone.zzlanshan.site.key;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;

    # 网站根目录
    root /www/wwwroot/andone.zzlanshan.site;
    index index.html;

    # 日志
    access_log /var/log/nginx/andone.zzlanshan.site.access.log;
    error_log /var/log/nginx/andone.zzlanshan.site.error.log;

    # 前端路由（Vue Router history模式）
    location / {
        try_files $uri $uri/ /index.html;
    }

    # API反向代理
    location /api/ {
        proxy_pass http://127.0.0.1:8000/api/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # 超时配置
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
    }

    # WebSocket反向代理（重要！）
    location /wss {
        proxy_pass http://127.0.0.1:8282;
        proxy_http_version 1.1;

        # WebSocket必需的头部
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";

        # 其他头部
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # 超时配置（WebSocket需要更长的超时时间）
        proxy_connect_timeout 7d;
        proxy_send_timeout 7d;
        proxy_read_timeout 7d;
    }

    # 静态文件缓存
    location ~* \.(jpg|jpeg|png|gif|ico|css|js|woff|woff2|ttf|svg)$ {
        expires 30d;
        add_header Cache-Control "public, immutable";
    }
}
```

#### 3.3 测试并重启Nginx

## 🎯 快速部署命令

### 一键部署脚本（服务器端）

```bash
#!/bin/bash
# deploy.sh - 快速部署脚本

# 配置变量
DOMAIN="andone.zzlanshan.site"
WEB_ROOT="/www/wwwroot/$DOMAIN"
BACKEND_ROOT="/www/wwwroot/andone.zzlanshan.site_backend"

# 1. 重启PHP-FPM
echo "重启PHP-FPM..."
systemctl restart php-fpm

# 2. 重启WebSocket服务
echo "重启WebSocket服务..."
cd $BACKEND_ROOT
php think im:server restart

# 3. 重启Nginx
echo "重启Nginx..."
nginx -t && systemctl restart nginx

# 4. 检查服务状态
echo "检查服务状态..."
systemctl status php-fpm --no-pager
ps aux | grep ImGateway | grep -v grep
systemctl status nginx --no-pager

echo "部署完成！"
echo "访问地址: https://$DOMAIN"
```

**使用方法**:
```bash
chmod +x deploy.sh
./deploy.sh
```
